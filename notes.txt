WHAT TO CREATE

    # net_util.py
    """Shared helpers for network-bound work: retries and parallel fan-out.

    Anything that calls yfinance / FRED / multpl should go through here so
    timeout, retry, and concurrency policies stay consistent.
    """

    from __future__ import annotations
    import logging
    import time
    from concurrent.futures import ThreadPoolExecutor
    from typing import Callable, Iterable, TypeVar

    log = logging.getLogger(__name__)

    T = TypeVar('T')
    R = TypeVar('R')

    def fetch_with_retry(fn: Callable[[], R],
                          retries: int = 3,
                          base_delay: float = 2.0) -> R:
        """Call fn up to `retries` times, exponential-ish back-off between
        attempts. Re-raises the last exception on final failure."""
        last_exc = None
        for attempt in range(retries):
            try:
                return fn()
            except Exception as e:
                last_exc = e
                if attempt < retries - 1:
                    time.sleep(base_delay * (attempt + 1))
        raise last_exc  # type: ignore[misc]

    def fetch_parallel(items: Iterable[T],
                        fn: Callable[[T], R],
                        max_workers: int = 6) -> dict[T, R]:
        """Map fn across items in a thread pool, return {item: result}.
        Exceptions from individual items become `None` in the result rather
        than aborting the whole batch."""
        items = list(items)
        workers = min(max_workers, max(1, len(items)))
        def _safe(x):
            try:
                return x, fn(x)
            except Exception as e:
                log.warning('fetch_parallel: %r failed: %s', x, e)
                return x, None
        with ThreadPoolExecutor(max_workers=workers) as pool:
            return dict(pool.map(_safe, items))

WHY
    Today every rendering callback has its own variation of:
        if not data: return html.P('Not ready yet…')
        try: ...
        except Exception as e: return html.Div(f'Error: {e}')
    That's ~15 copies of the same skeleton with subtly different wording.
    A decorator standardises the UX and removes the boilerplate.

WHAT TO CREATE

    # decorators.py
    """Error and loading boundaries for Dash render callbacks.

    Wrap any render callback whose job is to turn store data into children.
    The decorator catches `NotReadyError` and generic exceptions, returning a
    consistent placeholder instead of letting Dash show a 500."""

    from __future__ import annotations
    from functools import wraps
    import logging
    from dash import html
    from styles import CARD_MUTED, COLOR_MUTED, COLOR_WARN

    log = logging.getLogger(__name__)

    class NotReadyError(Exception):
        """Raised by a render callback when its input store isn't ready yet."""

    def _loading(msg: str):
        return html.Div(msg, style={**CARD_MUTED, 'color': COLOR_MUTED})

    def _error(label: str, err: Exception):
        return html.Div([
            html.P(f'{label} unavailable',
                   style={'color': COLOR_WARN, 'fontWeight': 600,
                          'margin': '0 0 6px'}),
            html.P(f'({type(err).__name__}: {err})',
                   style={'color': COLOR_MUTED, 'fontSize': '13px',
                          'margin': 0}),
        ], style=CARD_MUTED)

    def safe_render(label: str):
        """Decorator factory. `label` is shown in the error UI."""
        def outer(fn):
            @wraps(fn)
            def inner(*args, **kwargs):
                try:
                    return fn(*args, **kwargs)
                except NotReadyError as e:
                    return _loading(str(e) or 'Loading…')
                except Exception as e:
                    log.exception('safe_render %s failed', label)
                    return _error(label, e)
            return inner
        return outer

USAGE

    @app.callback(Output('summary', 'children'), Input('portfolio-data', 'data'))
    @safe_render('Summary')
    def update_summary(data):
        if not data:
            raise NotReadyError('Portfolio not loaded yet.')
